<script>
/* ─── STATE ─── */
let uploadTargetId = null;
let selectedFile   = null;
let allCategories  = [];

/* ─── UTILS ─── */
const $ = id => document.getElementById(id);
const baseUrl = () => $('baseUrl').value.replace(/\/$/, '');

function toast(msg, type = 'success') {
  const el = document.createElement('div');
  el.className = `toast ${type}`;
  el.innerHTML = `<span>${type === 'success' ? '✅' : '❌'}</span><span>${msg}</span>`;
  $('toast-root').appendChild(el);
  setTimeout(() => el.remove(), 3500);
}

function switchTab(name, btn) {
  document.querySelectorAll('.section').forEach(s => s.classList.remove('active'));
  document.querySelectorAll('.tab').forEach(t => t.classList.remove('active'));
  $('tab-' + name).classList.add('active');
  btn.classList.add('active');
}

function spinner() {
  return '<div class="loader-wrap"><div class="spinner"></div></div>';
}

function fmtSize(b) {
  if (b < 1024) return b + ' B';
  if (b < 1048576) return (b / 1024).toFixed(1) + ' KB';
  return (b / 1048576).toFixed(1) + ' MB';
}

/* ─── PRODUCTS ─── */
async function loadProducts() {
  $('products-area').innerHTML = spinner();
  try {
    const res = await fetch(`${baseUrl()}/items`);
    if (!res.ok) throw new Error('HTTP ' + res.status);
    const products = await res.json();
    renderProducts(products);
    toast(`${products.length} product(s) loaded`);
  } catch (e) {
    $('products-area').innerHTML = `<div class="empty"><div class="empty-icon">⚠️</div><h3>Failed to load</h3><p>${e.message}</p></div>`;
    toast('Failed: ' + e.message, 'error');
  }
}

function renderProducts(products) {
  if (!products.length) {
    $('products-area').innerHTML = `<div class="empty"><div class="empty-icon">📦</div><h3>No products found</h3><p>Add a product to get started.</p></div>`;
    return;
  }
  const cards = products.map(p => `
    <div class="product-card">
      <div class="product-img-wrap" id="img-${p.id}">
        <span class="no-img">🖼️</span>
      </div>
      <div class="product-body">
        <div class="product-id">ID · ${p.id ?? '—'}</div>
        <div class="product-name" title="${esc(p.name)}">${esc(p.name) || '—'}</div>
        <div class="product-desc">${esc(p.description) || 'No description'}</div>
        <div class="product-footer">
          <div class="product-price">${p.price != null ? '₹ ' + Number(p.price).toFixed(2) : '—'}</div>
          <button class="upload-mini" onclick="openModal(${p.id},'${esc(p.name)}')">📸 Image</button>
        </div>
      </div>
    </div>`).join('');
  $('products-area').innerHTML = `<div class="product-grid">${cards}</div>`;
  products.forEach(p => { if (p.id) fetchImage(p.id); });
}

function esc(str) {
  if (!str) return '';
  return str.replace(/&/g,'&amp;').replace(/</g,'&lt;').replace(/>/g,'&gt;').replace(/"/g,'&quot;').replace(/'/g,'&#39;');
}

async function fetchImage(itemId) {
  try {
    const res = await fetch(`${baseUrl()}/items/${itemId}/image`);
    if (!res.ok) return;
    const blob = await res.blob();
    if (!blob.size) return;
    const url = URL.createObjectURL(blob);
    const wrap = $(`img-${itemId}`);
    if (wrap) wrap.innerHTML = `<img src="${url}" alt="product"/>`;
  } catch (_) {}
}

/* ─── CATEGORIES ─── */
async function loadCategories() {
  $('categories-area').innerHTML = spinner();
  try {
    const res = await fetch(`${baseUrl()}/items/categories`);
    if (!res.ok) throw new Error('HTTP ' + res.status);
    allCategories = await res.json();
    renderCategories(allCategories);
    populateCatSelect();
    toast(`${allCategories.length} category/categories loaded`);
  } catch (e) {
    $('categories-area').innerHTML = `<div class="empty"><div class="empty-icon">⚠️</div><h3>Failed to load</h3><p>${e.message}</p></div>`;
    toast('Failed: ' + e.message, 'error');
  }
}

function renderCategories(cats) {
  if (!cats.length) {
    $('categories-area').innerHTML = `<div class="empty"><div class="empty-icon">🗂</div><h3>No categories</h3></div>`;
    return;
  }
  const rows = cats.map(c => `<tr><td class="cat-id">${c.categoryId ?? '—'}</td><td>${esc(c.name) || '—'}</td></tr>`).join('');
  $('categories-area').innerHTML = `
    <div class="cat-wrap">
      <table class="cat-table">
        <thead><tr><th>ID</th><th>Name</th></tr></thead>
        <tbody>${rows}</tbody>
      </table>
    </div>`;
}

function populateCatSelect() {
  const sel = $('f-category');
  sel.innerHTML = '<option value="">-- Select Category --</option>';
  allCategories.forEach(c => {
    const o = document.createElement('option');
    o.value = c.categoryId;
    o.textContent = `${c.name} (ID: ${c.categoryId})`;
    sel.appendChild(o);
  });
}

/* ─── SAVE PRODUCT ─── */
async function saveProduct() {
  const name = $('f-name').value.trim();
  if (!name) { toast('Product name is required', 'error'); return; }

  const payload = { name };
  const desc = $('f-desc').value.trim();
  const price = $('f-price').value;
  const catId = $('f-category').value;
  const itemId = $('f-id').value;
  if (desc)   payload.description = desc;
  if (price)  payload.price = parseFloat(price);
  if (catId)  payload.categoryId = parseInt(catId);
  if (itemId) payload.itemId = parseInt(itemId);

  const btn = $('save-btn');
  btn.disabled = true; btn.textContent = '⏳ Saving…';
  $('save-result').innerHTML = '';

  try {
    const res = await fetch(`${baseUrl()}/items/save`, {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify(payload)
    });
    const data = await res.json();
    if (res.ok && data.success !== false) {
      toast('Product saved! ✅');
      const id = data.item?.itemId ? ` · ID: ${data.item.itemId}` : '';
      $('save-result').innerHTML = `<div class="result-ok">✅ <strong>Saved!</strong> ${data.message || ''}${id}</div>`;
      clearForm();
    } else {
      throw new Error(data.message || 'Status ' + res.status);
    }
  } catch (e) {
    $('save-result').innerHTML = `<div class="result-err">❌ Failed: ${e.message}</div>`;
    toast('Save failed: ' + e.message, 'error');
  } finally {
    btn.disabled = false; btn.textContent = '💾 Save Product';
  }
}

function clearForm() {
  ['f-id','f-name','f-desc','f-price'].forEach(id => $(id).value = '');
  $('f-category').selectedIndex = 0;
}

/* ─── IMAGE UPLOAD MODAL ─── */
function openModal(itemId, itemName) {
  uploadTargetId = itemId;
  selectedFile = null;
  $('img-file').value = '';
  $('file-preview').classList.remove('show');
  $('upload-btn').disabled = true;
  $('modal-info').innerHTML = `<strong>Product:</strong> ${itemName} &nbsp;·&nbsp; <strong>ID:</strong> ${itemId}`;
  $('upload-modal').classList.add('show');
}

function closeModal() {
  $('upload-modal').classList.remove('show');
  uploadTargetId = null; selectedFile = null;
}

function onFileSelect(input) {
  const file = input.files[0];
  if (!file) return;
  selectedFile = file;
  $('preview-name').textContent = file.name;
  $('preview-size').textContent = fmtSize(file.size);
  const reader = new FileReader();
  reader.onload = e => { $('preview-img').src = e.target.result; };
  reader.readAsDataURL(file);
  $('file-preview').classList.add('show');
  $('upload-btn').disabled = false;
}

// Drag & drop
const dz = $('drop-zone');
dz.addEventListener('dragover', e => { e.preventDefault(); dz.classList.add('drag-over'); });
dz.addEventListener('dragleave', () => dz.classList.remove('drag-over'));
dz.addEventListener('drop', e => {
  e.preventDefault(); dz.classList.remove('drag-over');
  const file = e.dataTransfer.files[0];
  if (file && file.type.startsWith('image/')) {
    const dt = new DataTransfer(); dt.items.add(file);
    $('img-file').files = dt.files;
    onFileSelect($('img-file'));
  }
});

$('upload-modal').addEventListener('click', e => { if (e.target === $('upload-modal')) closeModal(); });

async function doUpload() {
  if (!selectedFile || !uploadTargetId) return;
  const btn = $('upload-btn');
  btn.disabled = true; btn.textContent = '⏳ Uploading…';

  const form = new FormData();
  form.append('file', selectedFile);

  try {
    const res = await fetch(`${baseUrl()}/items/${uploadTargetId}/upload-image`, {
      method: 'POST', body: form
    });
    const data = await res.json();
    if (res.ok && data.success !== false) {
      toast('Image uploaded! 🖼️');
      closeModal();
      fetchImage(uploadTargetId); // refresh card image
    } else {
      throw new Error(data.message || 'Status ' + res.status);
    }
  } catch (e) {
    toast('Upload failed: ' + e.message, 'error');
  } finally {
    btn.disabled = false; btn.textContent = '⬆️ Upload';
  }
}
